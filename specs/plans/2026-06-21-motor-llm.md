# MOTOR-LLM Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** The AI proposes its OWN NBA hypothesis (LLM, text), a deterministic SQL analysis falsifies/confirms it, and the AI acts alone within the human-approved range (non-money) or escalates honestly — learning across ≤3 loops, with the token cost of each decision visible.

**Architecture:** A new `server/motor/` module drives the EXISTING CP2 floor. The LLM is the `MotorReasoning` provider the `02:1A` seam already anticipated — it only picks an `action_code` that `fn_nba_test_all` already returned (§8: never a number) + a `[C]` confidence + reasoning. `runMotor` runs the ≤3 loop per restaurant: `proposeHypothesis` → `validateHypothesis` (confirmed gap? in `allowed_today`?) → ACT via `proposeNba` (LLM-chosen lever adapter) + `autoDispatch` (the authoritative `auto_releasable`/money gate, reused unchanged) OR escalate. Each terminal outcome writes one `Knowledge_Case` (`reviewed=false`, human-gated). Zero new tables; `motor_attempt_id` ties cost (`Llm_Usage_Log`) + learning (`Knowledge_Case`) + action (`Decision_Trace` origin=auto).

**Tech Stack:** TS strict, tRPC v11 + Express, React 19 + wouter, shadcn/Tailwind, Zod v3, Supabase Postgres, Vitest/pgTAP/Playwright, OpenAI gpt-4o-mini (existing `ChatClient`).

**Invariants (every task obeys; pinned by tests):** number always SQL (§14/§8 — LLM never a number, only picks an existing `action_code` + text) · tenant server-side, cohort presence ≠ ownership (§3.4) · RESULT NULL pre-run · knobs by name (§3.8) · ≤100 lines/unit · reuse before create · TDD red→green (anti-fake FIRST) · 1 commit/task · `autoDispatch`/`sealMinCalculationNBA` are the SOLE `auto_releasable` authority (no gate re-implementation — §3.11) · Codex review per phase.

// server/motor/reasoning.ts
// 02C — the LLM provider the 02:1A seam anticipated. It picks ONE action_code that fn_nba_test_all ALREADY
// returned (§8: never invents a number) + a [C] confidence + reasoning. The deterministic stub keeps CI
// reproducible. leverAdapter lets the existing proposeNba write the LLM-chosen lever (reuse §3.11).
import type { NbaVerdict, NbaReasoning, NbaSelection } from "../agente/reasoning.js";
import { labelOf, fmtValue, fmtGap } from "../../shared/signalFormat.js";

export interface MotorHypothesis {
  lever: NbaVerdict | null;   // null ⇒ no confident in-range candidate ⇒ escalate (no-suppose)
  rootCause: string;          // [C] bounded by the verdict, never asserts a number SQL didn't return
  confidence: number | null;  // [C] self-confidence 0..1; null ⇒ escalate
  reasoning: string;          // [C] the /problem-solving + /sat narrative
}

export interface MotorReasoning {
  proposeHypothesis(input: {
    verdicts: NbaVerdict[];
    discarded: { action_code: string; reason: string }[];
    grounding: { pattern: string; outcome: string; not_resolved_reason: string | null }[];
  }): Promise<MotorHypothesis>;
}

const severity = (v: NbaVerdict): number =>
  v.standard === 0 ? Infinity : Math.abs(v.gap!) / Math.abs(v.standard!);

// Deterministic stand-in for the LLM: worst-relative-gap problem lever not already falsified. Same selection
// math as deterministicReasoning, but honoring `discarded` so the ≤3 loop converges. confidence fixed (a stub
// can't reason); the REAL provider returns a genuine [C].
export const stubMotorReasoning: MotorReasoning = {
  async proposeHypothesis({ verdicts, discarded }) {
    const dead = new Set(discarded.map((d) => d.action_code));
    const problems = verdicts
      .filter((v) => (v.verdict === "below" || v.verdict === "above") && v.gap != null && !dead.has(v.action_code))
      .sort((a, b) => (severity(a) === severity(b) ? a.action_code.localeCompare(b.action_code) : severity(b) - severity(a)));
    const lever = problems[0] ?? null;
    if (!lever) return { lever: null, rootCause: "no remaining in-range gap", confidence: null, reasoning: "all levers ok or falsified" };
    const dim = lever.dimension ?? "signal";
    return {
      lever,
      rootCause: `${labelOf(dim)} ${lever.verdict} standard — ${fmtValue(dim, lever.measured!)} vs ${fmtValue(dim, lever.standard!)} (gap ${fmtGap(dim, lever.gap!)})`,
      confidence: 0.9,
      reasoning: "stub: worst-relative-gap heuristic",
    };
  },
};

// Wrap a chosen lever as an NbaReasoning so proposeNba writes THAT lever's proposal (no proposeNba edit).
export const leverAdapter = (lever: NbaVerdict, rootCause: string): NbaReasoning => ({
  select(): NbaSelection {
    return { ranked: [lever.action_code], lever, rootCause };
  },
});
```

// server/motor/validateHypothesis.ts
// 02C — the DETERMINISTIC falsifier. `confirmed` = the lever is a real SQL problem (below/above with a gap —
// fn_nba_test already computed it; we read its fields, never recompute a number). `inRange` = the action is in
// the human-signed Policy_Tier.allowed_today.auto_actions (the approved range). The auto_releasable/money gate
// is NOT re-implemented here (§3.11): the SOLE authority is sealMinCalculationNBA (in proposeNba) +
// autoDispatch's DB re-check. This advisory pre-check only decides retry-vs-escalate-vs-attempt.
import type pg from "pg";
import type { NbaVerdict } from "../agente/reasoning.js";

export interface Validation {
  confirmed: boolean; // SQL says this lever is out of range with a real gap
  inRange: boolean;   // action_code ∈ allowed_today.auto_actions for the cohort's tier
}

export async function validateHypothesis(
  lever: NbaVerdict,
  _restaurantId: string,
  _cohortId: string,
  tierId: string,
  client: pg.PoolClient,
): Promise<Validation> {
  const confirmed = (lever.verdict === "below" || lever.verdict === "above") && lever.gap != null;
  const r = await client.query<{ ok: boolean }>(
    `select coalesce(allowed_today->'auto_actions' @> to_jsonb($2::text), false) as ok
       from gov."Policy_Tier" where tier_id = $1
       order by policy_version desc limit 1`,
    [tierId, lever.action_code],
  );
  return { confirmed, inRange: r.rows[0]?.ok ?? false };
}
```

```ts
// server/motor/learn.ts
// 02C — the learning store. Each terminal decision writes ONE Knowledge_Case (reviewed=false ⇒ does NOT
// ground future runs until a human approves it in the controls panel — the BR-B16 RL-guard). Grounding reads
// ONLY reviewed cases (so the AI never learns from un-vetted text). Numbers are never written here (§14):
// outcome is a measured RESULT [V], the narrative is [C].
import type pg from "pg";

export interface MotorCase {
  tenantId: string;
  areaType: string;                  // lever.dimension
  pattern: string;                   // `${dimension}_${verdict}`
  outcome: "resolved" | "escalated"; // resolved = AI acted alone; escalated = handed to human
  resolution?: string | null;        // [C] what it did (acted)
  notResolvedReason?: string | null; // [C] why escalated
  discarded: { action_code: string; reason: string }[];
  attemptId: string;                 // motor_attempt_id — ties cost (Llm_Usage_Log.ref_id)
  nbaId: string | null;              // the dispatched NBA when acted
}

export async function writeMotorCase(k: MotorCase, client: pg.PoolClient): Promise<string> {
  const prov = JSON.stringify({ outcome: "[V]", resolution: "[C]", not_resolved_reason: "[C]" });
  const path = JSON.stringify({ attempt_id: k.attemptId, nba_id: k.nbaId, resolution: k.resolution ?? null });
  const r = await client.query<{ kb_case_id: string }>(
    `insert into tenant."Knowledge_Case"
       (tenant_id, area_type, pattern, outcome, resolution, path_used, not_resolved_reason, discarded_branches, reviewed, provenance_by_field)
     values ($1,$2,$3,$4,$5,$6::jsonb,$7,$8::jsonb,false,$9::jsonb)
     returning kb_case_id`,
    [k.tenantId, k.areaType, k.pattern, k.outcome, k.resolution ?? null, path, k.notResolvedReason ?? null, JSON.stringify(k.discarded), prov],
  );
  return r.rows[0]!.kb_case_id;
}

export async function readGrounding(
  tenantId: string,
  areas: string[],
  client: pg.PoolClient,
): Promise<{ pattern: string; outcome: string; not_resolved_reason: string | null }[]> {
  const r = await client.query<{ pattern: string; outcome: string; not_resolved_reason: string | null }>(
    `select pattern, outcome, not_resolved_reason from tenant."Knowledge_Case"
      where tenant_id=$1 and reviewed=true and outcome is not null and area_type = any($2)
      order by created_at desc limit 8`,
    [tenantId, areas],
  );
  return r.rows;
}
```

// server/motor/runMotor.ts
import { randomUUID } from "node:crypto";
import type pg from "pg";
import { query, withTx } from "../db/pool.js";
import { knobNum } from "../_core/knobs.js"; // helper that reads catalog.knob_required_num by name (create if absent — see note)
import { proposeNba } from "../agente/nba_engine.js";
import { autoDispatch } from "../cockpit/autoDispatch.js";
import { type MotorReasoning, leverAdapter } from "./reasoning.js";
import { validateHypothesis } from "./validateHypothesis.js";
import { writeMotorCase, readGrounding } from "./learn.js";

export interface MotorAttemptInput { restaurantId: string; cohortId: string; week: string; tenantId: string; tierId: string; }
export interface MotorAttemptResult { outcome: "acted" | "escalated"; nbaId: string | null; loops: number; attemptId: string; }

// 02C — one restaurant, the ≤3 loop. The AI proposes → SQL falsifies/confirms → ACT (in-range non-money) or
// ESCALATE (out-of-range / no-confident / exhausted). The auto_releasable/money authority is the seal +
// autoDispatch (reused). Every iteration's tokens are recorded by the provider under ref_id=attemptId.
export async function runMotorAttempt(
  i: MotorAttemptInput, reasoning: MotorReasoning, pool: pg.Pool,
): Promise<MotorAttemptResult> {
  const attemptId = randomUUID();
  const maxLoops = await knobNum("motor_max_loops");
  const minConf = await knobNum("motor_min_confidence");
  const verdicts = await query(
    `select action_code, dimension, measured::float8 measured, standard::float8 standard, verdict,
            gap::float8 gap, within_range, n_min_ok, k_anon_ok from cohort.fn_nba_test_all($1,$2)`,
    [i.restaurantId, i.week],
  ) as never[];
  const areas = [...new Set((verdicts as { dimension: string | null }[]).map((v) => v.dimension).filter(Boolean) as string[])];
  const grounding = await withTx((c) => readGrounding(i.tenantId, areas, c));
  const discarded: { action_code: string; reason: string }[] = [];

  for (let k = 0; k < maxLoops; k++) {
    const hyp = await reasoning.proposeHypothesis({ verdicts, discarded, grounding });
    if (!hyp.lever || hyp.confidence == null || hyp.confidence < minConf) {
      return escalate(i, attemptId, "no_confident_hypothesis", discarded, k + 1);
    }
    const lever = hyp.lever;
    const val = await withTx((c) => validateHypothesis(lever, i.restaurantId, i.cohortId, i.tierId, c));
    if (!val.confirmed) { discarded.push({ action_code: lever.action_code, reason: "sql_no_gap" }); continue; }
    if (!val.inRange) {
      discarded.push({ action_code: lever.action_code, reason: "out_of_range" });
      return escalate(i, attemptId, "out_of_range", discarded, k + 1);
    }
    // ATTEMPT ACT — proposeNba seals the AUTHORITATIVE auto_releasable; autoDispatch re-checks §7.
    const res = await withTx((c) => proposeNba({ restaurantId: i.restaurantId, cohortId: i.cohortId, week: i.week }, leverAdapter(lever, hyp.rootCause), c));
    const gate = (await query<{ ar: boolean | null; fc: string | null }>(
      `select m.auto_releasable as ar, p.financial_class::text as fc from gov."NBA_Proposal" p
         left join lateral (select auto_releasable from gov."min_calculation" where nba_id=p.nba_id::text order by computed_at desc limit 1) m on true
        where p.nba_id=$1::uuid`, [res.nbaId]))[0];
    if (gate?.ar === true && gate.fc !== "direct") {
      try {
        await withTx((c) => autoDispatch(res.nbaId, i.tenantId, c));
        await withTx((c) => writeMotorCase({ tenantId: i.tenantId, areaType: lever.dimension ?? "signal", pattern: `${lever.dimension}_${lever.verdict}`, outcome: "resolved", resolution: hyp.rootCause, discarded, attemptId, nbaId: res.nbaId }, c));
        return { outcome: "acted", nbaId: res.nbaId, loops: k + 1, attemptId };
      } catch { return escalate(i, attemptId, "dispatch_failed", discarded, k + 1); }
    }
    return escalate(i, attemptId, "gate_failed_at_seal", discarded, k + 1);
  }
  return escalate(i, attemptId, "exhausted_3_loops", discarded, maxLoops);
}

async function escalate(i: MotorAttemptInput, attemptId: string, reason: string, discarded: { action_code: string; reason: string }[], loops: number): Promise<MotorAttemptResult> {
  await withTx((c) => writeMotorCase({ tenantId: i.tenantId, areaType: discarded[0]?.action_code.split("_")[0] ?? "signal", pattern: reason, outcome: "escalated", notResolvedReason: reason, discarded, attemptId, nbaId: null }, c));
  return { outcome: "escalated", nbaId: null, loops, attemptId };
}
